def initialize(key, value, hash = nil, ttl = 3600)
  hash = self.class.hashify_key(key) if hash.blank?
  self.key = key
  self.hash = hash
  self.value = value
  self.ttl = ttl
  self.refresh
end

def valid?(provider = nil)
  provider ||= Elephas::Cache.provider
  provider.now - self.updated_at < self.ttl
end
