def delete(key)
  fkey = key.ensure_string
  rv = @data.has_key?(fkey)
  @data.delete(fkey)
  rv
end

def exists?(key)
  fkey = key.ensure_string
  @data.has_key?(fkey) && @data[fkey].valid?
end

def write(key, value, options = {})
  fvalue = ::Elephas::Entry.ensure(value, key, options)
  fvalue.refresh
  @data[key.ensure_string] = fvalue
  value
end
